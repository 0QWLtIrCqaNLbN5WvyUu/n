		html += '	<span class="badge thank" style="display:none">';
		html += '		<a href="#thankBadgeModal" class="popup-modal">';
		html += '			<figure><img src="common/img/bg-badge-thank.svg" alt=""/></figure>	';

		html += '	<span class="gbValue"><img style="display:none;" class="icon-goodbad"/></span>';
		html += '	</h3>';
		html += '	<div class="tx"></div>';
		html += '	<div class="goodBad">';
		html += '		<ul>';
		html += '			<li class="good">';
					html += '				<div class="btn"><a href="#btn_good_comment" k_id=""><img src="common/img/btn-good.png" width="100" height="41" alt="GOOD"/></a></div>';
				html += '				<div class="right">';
		html += '					<span class="count">0</span>';
		html += '					<span class="unit">点</span>';
		html += '				</div>';
		html += '			</li>';
		html += '			<li class="bad">';
					html += '				<div class="btn"><a href="#btn_bad_comment" k_id=""><img src="common/img/btn-bad.png" width="100" height="41" alt="BAD"/></a></div>';
				html += '				<div class="right">';
		html += '					<span class="count">0</span>';
		html += '					<span class="unit">点</span>';
		html += '				</div>';
		html += '			</li>';
		html += '			<li class="users" style="display:none">';
		html += '				<a href="#open_usersModal" k_id=""><img src="common/img/icon-users.svg" alt=""/></a>';
		html += '			</li>';
		html += '			<li class="reply">';
					html += '				<a href="#btn_reply_comment" k_id=""><div class="repBtn">返信</div></a>';
				html += '			</li>';
		html += '		</ul>';
		html += '		<a href="#doInfoModal2"><div class="outBtn">違反報告</div></a>';
		html += '	</div>';
		html += '</li>';
		for(let i in resp){
			let kuchikomi_comment = resp[i];
			let $parent = $('section[kuchikomi_id="'+kuchikomi_comment.kuchikomi_id+'"].Report');
			let $commentBlock = $parent.find('.userComBlock');
			if($commentBlock.length == 0)continue;
			let $ul = $commentBlock.find('ul').empty();
			if(kuchikomi_comment.comments.length <= 2)
			{
				if($parent.find('.grad-trigger').length > 0){
					$parent.find('.grad-trigger').remove();
					$parent.find('.grad-btn').remove();
					$commentBlock.unwrap().unwrap();
				}
			}
			for(let j in kuchikomi_comment.comments){
				let comment = kuchikomi_comment.comments[j];
				let $dom = $(html);
				if(comment.action == '1'){
					$dom.find('.icon-goodbad').attr('src', 'common/img/icon-good.svg').show();
				}else if(comment.action == '2'){
					$dom.find('.icon-goodbad').attr('src', 'common/img/icon-bad.svg').show();
				}
				$dom.find('.good .count').text(comment.good);
				$dom.find('.bad .count').text(comment.bad);
				$dom.find('[href="#open_usersModal"]').attr({k_id: comment.kuchikomi_id, comment_id:comment.comment_id});
				$dom.attr({comment_id:comment.comment_id});
				if(comment.member_id){
					$dom.find('a.name').attr('href', './userpage.php?user_id='+comment.member_id);
				}
				if(comment.comment_no){
					$dom.find('span.comment_no').text(comment.comment_no+'. ');
				}
				if(comment.good || comment.bad){
					$dom.find('.users').show();
				}
				$dom.find('a.name').text(comment.member_name+'さん');
				$dom.find('div.tx').html(comment.comment);
				if(comment.comment == '' && comment.mosaic){
					$dom.find('div.tx').html('<img width="770" height="166" alt="モザイク" src="'+comment.mosaic+'">');
					$dom.find('li.good, li.bad, li.users, [href="#doInfoModal2"]').hide();
					$dom.find('[href="#btn_reply_comment"]').hide();
					$dom.find('[href="#commentBanModal"]').hide();
				}
				$dom.find('a[href="#doInfoModal2"]').on('click', on_info_modal);
				$dom.find('[k_id]').attr('k_id', comment.kuchikomi_id);
				if(comment.member_recv_thank){
					$dom.find('.badge').css('display', 'inline-block').find('.num').text(comment.member_recv_thank);
				}
				if(comment.member_paid){
					$dom.find('.paid').show();
				}
				if(comment.member_type == 2){
					$dom.find('.unei').show();
				}
				$ul.append($dom);
			}

			if(kuchikomi_comment.comments.length == 0){
				$commentBlock.hide();
			}else{
				$commentBlock.show();
			}
		}
		initMagnificPopup2();
	});
}

$('button[name="submit_btn"]').on('click', function(){
	let $parent = $(this).closest('section.Report');
	let data = {};
	data.kuchikomi_id = $parent.attr('kuchikomi_id');
	data.comment = $parent.find('textarea[name="comment"]').val() || '';
	data.goodbad = $('#ac-'+data.kuchikomi_id).attr('goodbad');
	data.comment_id = $('#ac-'+data.kuchikomi_id).attr('comment_id');
	data.goodbad_id = $('#ac-'+data.kuchikomi_id).attr('goodbad_id');
	if(data.comment == '')return;

	$parent.find('textarea[name="comment"]').val('');
	$parent.find('.ac-box label').trigger('click');

		type: 'POST',
		url: 'ajax_comment.php?submit',
		data: data,
		cache: false,
		dataType: 'json',
	}).done(function(resp){
		load_comments();
	});
});
